3.2 Scatter Plot

data %>% 
  filter(lg_id == "AL") %>% 
  mutate(team_id = fct_lump(team_id, n = 12)) %>% 
  filter(team_id != "Other") %>% 
  filter(complete.cases(.)) %>%
  filter(year_id>1990) %>% 
  group_by(team_id,year_id) %>% 
  summarise_at(vars(g:gidp), ~sum(.)) %>% 
  ggplot(aes(x=year_id,y=hr,color = fct_reorder(team_id,-hr))) +
  geom_point()
