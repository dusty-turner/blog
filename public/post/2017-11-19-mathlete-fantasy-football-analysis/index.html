Clean and Parse

Since the data is listed in many rows and columns, we need to do lots of work:

# change data types from factors to characters


fantasy = 


fantasy %>%


  mutate(Result = as.character(Result)) %>%


  mutate(Home = as.character(Home)) %>%


  mutate(Away = as.character(Away)) %>%


  filter(str_detect(Result, "-")) 




# split the score column on the '-'


scoresvec = unlist(strsplit(fantasy$Result, "-"))




# put every other value in an away and home column


fantasy$awayscore = scoresvec[c(TRUE,FALSE)]


fantasy$homescore = scoresvec[c(FALSE,TRUE)]




# get rid of the records from the end of the team names


fantasy$Away = substr(fantasy$Away,1,nchar(fantasy$Away)-6)


fantasy$Home = substr(fantasy$Home,1,nchar(fantasy$Home)-6)

Getting better. Now lets organize the data in 3 columns: Team Name, Score, and Who Scored The Points

##create df


fantasy = 


fantasy %>%


select(Away, awayscore) %>%


  mutate(Team = Away) %>%


  mutate(Score = awayscore) %>%


  mutate(Who = "For") %>%


  select(Team, Score, Who) %>%


  bind_rows(


    fantasy %>%


      select(Home, homescore) %>%


      mutate(Team = Home) %>%


      mutate(Score = homescore) %>%


      mutate(Who = "For") %>%


      select(Team, Score, Who)


  ) %>%


  bind_rows(


    fantasy %>%


      select(Away, homescore) %>%


      mutate(Team = Away) %>%


      mutate(Score = homescore) %>%


      mutate(Who = "Against") %>%


      select(Team, Score, Who) 


   ) %>%


  bind_rows(


    fantasy %>%


      select(Home, awayscore) %>%


      mutate(Team = Home) %>%


      mutate(Score = awayscore) %>%


      mutate(Who = "Against") %>%


      select(Team, Score, Who)


  )
