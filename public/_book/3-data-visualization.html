animdata =
  data %>%
  filter(!is.na(lg_id)) %>%
  group_by(year_id, team_id) %>%
  summarise_at(vars(hr, so, bb, ab), ~ sum(., na.rm = TRUE)) %>%
  ungroup() %>%
  inner_join(data %>%
               select(year_id, team_id, lg_id) %>%
               distinct())
